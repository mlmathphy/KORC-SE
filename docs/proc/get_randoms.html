  SUBROUTINE get_randoms(nums)
    USE omp_lib
    IMPLICIT NONE

    REAL(rp), DIMENSION(:), INTENT(OUT) :: nums

    INTEGER                             :: i

    !$OMP PARALLEL DO DEFAULT(SHARED) PRIVATE(i)
    DO i = 1, SIZE(nums)
       nums(i) = get_random()
    END DO
    !$OMP END PARALLEL DO
  END SUBROUTINE get_randoms
