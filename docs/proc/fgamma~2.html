  FUNCTION fGamma(x,k,t)
    REAL(rp), INTENT(IN) :: x ! Independent variable
    REAL(rp), INTENT(IN) :: k ! Shape factor
    REAL(rp), INTENT(IN) :: t ! Scale factor
    REAL(rp) :: fGamma

    fGamma = x**(k - 1.0_rp)*EXP(-x/t)/(GAMMA(k)*t**k)
  END FUNCTION fGamma
